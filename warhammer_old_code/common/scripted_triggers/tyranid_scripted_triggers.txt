is_tyranid_hive = {
	OR = {
		is_planet_class = pc_tyranid_infected
		is_planet_class = pc_tyranid_infested
		is_planet_class = pc_tyranid_hive
	}
}

has_tyranid_hive_flag = {
	OR = {
		has_planet_flag = tyranid_incubation
		has_planet_flag = tyranid_infected
		has_planet_flag = tyranid_infested
		has_planet_flag = tyranid_hive
	}
}
